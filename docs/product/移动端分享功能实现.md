## 二、Web Share API 实现

### 2.1 原生分享 API

```javascript
// share.js - 分享功能模块

class ShareManager {
  constructor() {
    this.isSupported = 'share' in navigator;
    this.platforms = {
      wechat: 'wx://',
      weibo: 'weibo://',
      qq: 'mqqapi://'
    };
  }

  // 原生分享
  async nativeShare(data) {
    if (this.isSupported) {
      try {
        const shareData = {
          title: data.title || document.title,
          text: data.description || document.querySelector('meta[name="description"]')?.content || '',
          url: data.url || window.location.href
        };

        await navigator.share(shareData);
        return { success: true };
      } catch (err) {
        if (err.name === 'AbortError') {
          return { success: false, reason: 'user_cancelled' };
        }
        console.error('Share failed:', err);
        return { success: false, reason: err.message };
      }
    }
    return { success: false, reason: 'not_supported' };
  }

  // 检查分享支持
  canNativeShare() {
    return this.isSupported;
  }
}

const shareManager = new ShareManager();
```

## 三、分享平台 URL 方案

### 3.1 社交平台分享链接

```javascript
// 社交平台分享链接生成
const ShareLinks = {
  // 微信分享 (唤起微信)
  wechat: (url, title) => {
    return `weixin://dl/webpage?url=${encodeURIComponent(url)}`;
  },

  // 微博分享
  weibo: (url, title, description) => {
    return `https://service.weibo.com/share/share.php?url=${encodeURIComponent(url)}&title=${encodeURIComponent(title + ' ' + description)}`;
  },

  // 知乎分享
  zhihu: (url, title) => {
    return `https://www.zhihu.com/share/article?url=${encodeURIComponent(url)}&title=${encodeURIComponent(title)}`;
  },

  // QQ分享
  qq: (url, title, description) => {
    return `https://connect.qq.com/widget/shareqq/iframe_index.html?url=${encodeURIComponent(url)}&title=${encodeURIComponent(title)}&desc=${encodeURIComponent(description)}`;
  },

  // QQ空间分享
  qzone: (url, title, description, pics) => {
    return `https://sns.qzone.qq.com/cgi-bin/qzshare/cgi_qzshare_onekey?url=${encodeURIComponent(url)}&title=${encodeURIComponent(title)}&desc=${encodeURIComponent(description)}&summary=${encodeURIComponent(description)}&pics=${pics || ''}`;
  },

  // 推特分享
  twitter: (url, title) => {
    return `https://twitter.com/intent/tweet?url=${encodeURIComponent(url)}&text=${encodeURIComponent(title)}`;
  },

  // 领英分享
  linkedin: (url, title) => {
    return `https://www.linkedin.com/sharing/share-offsite/?url=${encodeURIComponent(url)}`;
  }
};
```

### 3.2 分享函数实现

```javascript
// 分享函数
function getShareData() {
  const title = document.querySelector('meta[property="og:title"]')?.content ||
                document.querySelector('h1')?.textContent ||
                document.title;
  const description = document.querySelector('meta[property="og:description"]')?.content ||
                      document.querySelector('meta[name="description"]')?.content ||
                      '';
  const url = window.location.href;
  const image = document.querySelector('meta[property="og:image"]')?.content || '';

  return { title, description, url, image };
}

function shareToWechat() {
  const data = getShareData();
  // 微信分享需要使用自定义引导
  showWechatQR(data.url);
  closeSharePanel();
}

function shareToWeibo() {
  const data = getShareData();
  window.open(ShareLinks.weibo(data.url, data.title, data.description), '_blank');
  closeSharePanel();
}

function shareToZhihu() {
  const data = getShareData();
  window.open(ShareLinks.zhihu(data.url, data.title), '_blank');
  closeSharePanel();
}

function shareToQQ() {
  const data = getShareData();
  window.open(ShareLinks.qq(data.url, data.title, data.description), '_blank');
  closeSharePanel();
}

function copyLink() {
  const data = getShareData();
  navigator.clipboard.writeText(data.url).then(() => {
    showToast('链接已复制到剪贴板');
    closeSharePanel();
  }).catch(() => {
    // Fallback
    const textarea = document.createElement('textarea');
    textarea.value = data.url;
    document.body.appendChild(textarea);
    textarea.select();
    document.execCommand('copy');
    document.body.removeChild(textarea);
    showToast('链接已复制到剪贴板');
    closeSharePanel();
  });
}

// 分享面板控制
function openSharePanel() {
  document.getElementById('share-panel')?.classList.add('open');
  document.body.style.overflow = 'hidden';
}

function closeSharePanel() {
  document.getElementById('share-panel')?.classList.remove('open');
  document.body.style.overflow = '';
}

// Toast 提示
function showToast(message) {
  const toast = document.createElement('div');
  toast.className = 'share-toast';
  toast.textContent = message;
  document.body.appendChild(toast);

  setTimeout(() => toast.classList.add('show'), 10);
  setTimeout(() => {
    toast.classList.remove('show');
    setTimeout(() => toast.remove(), 300);
  }, 2000);
}
```

```javascript
function showWechatQR(url) {
  const guide = document.getElementById('wechat-guide');
  const qrContainer = document.getElementById('wechat-qrcode');

  // 使用 QRCode.js 生成二维码
  qrContainer.innerHTML = '';
  new QRCode(qrContainer, {
    text: url,
    width: 200,
    height: 200,
    colorDark: '#000000',
    colorLight: '#ffffff',
    correctLevel: QRCode.CorrectLevel.M
  });

  guide.style.display = 'flex';
}

function closeWechatGuide() {
  document.getElementById('wechat-guide').style.display = 'none';
}
```
