## 二、Google Analytics 4 配置

### 2.1 创建 GA4 属性

1. 访问 [Google Analytics](https://analytics.google.com/)
2. 点击「管理」→「创建属性」
3. 选择「GA4」属性类型
4. 填写属性名称：`shenhuanjie.github.io`
5. 设置时区和货币

### 2.2 获取 Measurement ID

1. 在 GA4 属性中，点击「数据流」
2. 添加 Web 数据流
3. 填写网站 URL：`https://shenhuanjie.github.io`
4. 获取 **Measurement ID**（格式：`G-XXXXXXXXXX`）

### 2.3 配置方式对比

#### 方式一：Chic 主题配置（推荐）

编辑 `themes/Chic/_config.yml`：

```yaml
# Google Analytics 4
google_analytics:
  enable: true
  tracking_id: G-XXXXXXXXXX  # 替换为您的 GA4 测量 ID
```

#### 方式二：根 _config.yml 配置

编辑项目根目录 `_config.yml`：

```yaml
google_analytics:
  enable: true
  measurement_id: G-XXXXXXXXXX
  options:
    anonymize_ip: true
    send_page_view: true
    debug_mode: false
```

### 2.4 手动添加 GA4 代码

如果主题不直接支持 GA4 配置，可在 `themes/chic/layout/_partial/head.ejs` 中添加：

```html
<!-- Google tag (gtag.js) -->
<script async src="https://www.googletagmanager.com/gtag/js?id=G-XXXXXXXXXX"></script>
<script>
  window.dataLayer = window.dataLayer || [];
  function gtag(){dataLayer.push(arguments);}
  gtag('js', new Date());

  // 配置参数
  gtag('config', 'G-XXXXXXXXXX', {
    'anonymize_ip': true,
    'send_page_view': true
  });
</script>
```

## 三、事件追踪配置（v2.2 Phase 3 新增）

### 3.1 核心事件列表

| 事件名称 | 触发条件 | 事件参数 | 用途 |
|----------|----------|----------|------|
| `article_read_complete` | 阅读进度 > 90% | article_title, category | 文章完成率分析 |
| `article_read_progress` | 每 25% 进度 | article_title, progress (25/50/75) | 阅读深度分析 |
| `share_click` | 点击分享按钮 | platform (weixin/zhihu/twitter) | 分享渠道分析 |
| `search_query` | 提交搜索 | search_term | 搜索关键词分析 |
| `external_link_click` | 点击外部链接 | link_url, link_text | 跳出链接分析 |
| `toc_navigate` | 点击目录链接 | heading_id, heading_text | 目录使用分析 |

### 3.2 事件追踪代码实现

#### 阅读完成率追踪

在 `themes/Chic/layout/_partial/head.ejs` 或文章详情模板中添加：

```html
<script>
// 文章阅读完成率追踪 (v2.2 Phase 3)
(function() {
  let trackedSections = new Set();
  let scrollPercent = 0;

  function getArticleProgress() {
    const scrollHeight = document.documentElement.scrollHeight - window.innerHeight;
    return scrollHeight > 0 ? (window.scrollY / scrollHeight) * 100 : 0;
  }

  function trackEvent(name, params) {
    if (typeof gtag !== 'undefined') {
      gtag('event', name, {
        event_category: 'engagement',
        ...params
      });
    }
  }

  // 节流函数
  function throttle(func, limit) {
    let inThrottle;
    return function(...args) {
      if (!inThrottle) {
        func.apply(this, args);
        inThrottle = true;
        setTimeout(() => inThrottle = false, limit);
      }
    };
  }

  // 滚动监听
  window.addEventListener('scroll', throttle(function() {
    scrollPercent = getArticleProgress();

    // 阅读完成（滚动超过 90%）
    if (scrollPercent > 90) {
      if (!trackedSections.has('complete')) {
        trackedSections.add('complete');
        trackEvent('article_read_complete', {
          event_label: document.title,
          value: Math.round(scrollPercent)
        });
      }
    }

    // 阅读进度追踪（25%, 50%, 75%）
    [25, 50, 75].forEach(threshold => {
      if (scrollPercent >= threshold && scrollPercent < threshold + 5) {
        const key = 'progress_' + threshold;
        if (!trackedSections.has(key)) {
          trackedSections.add(key);
          trackEvent('article_read_progress', {
            event_label: document.title,
            value: threshold
          });
        }
      }
    });
  }, 1000));
})();
</script>
```

#### 分享按钮点击追踪

```html
<script>
// 分享按钮点击追踪 (v2.2 Phase 3)
(function() {
  function trackShareClick(platform) {
    if (typeof gtag !== 'undefined') {
      gtag('event', 'share_click', {
        event_category: 'social',
        event_label: platform,
        value: 1
      });
    }
  }

  // 监听分享按钮点击
  document.querySelectorAll('.share-button, .jiathis_button_weixin, .AddToShareBar a').forEach(function(button) {
    button.addEventListener('click', function(e) {
      // 尝试识别分享平台
      var platform = 'unknown';
      var classes = this.className.toLowerCase();
      var text = this.textContent.toLowerCase();

      if (classes.includes('weixin') || text.includes('微信')) {
        platform = 'weixin';
      } else if (classes.includes('zhihu') || text.includes('知乎')) {
        platform = 'zhihu';
      } else if (classes.includes('twitter') || text.includes('twitter')) {
        platform = 'twitter';
      } else if (classes.includes('weibo') || text.includes('微博')) {
        platform = 'weibo';
      }

      trackShareClick(platform);
    });
  });
})();
</script>
```

#### 外部链接点击追踪

```html
<script>
// 外部链接点击追踪 (v2.2 Phase 3)
(function() {
  function isExternalLink(url) {
    return url && (url.startsWith('http://') || url.startsWith('https://'))
      && !url.includes(window.location.hostname);
  }

  document.querySelectorAll('a[href]').forEach(function(link) {
    link.addEventListener('click', function(e) {
      var url = this.getAttribute('href');
      if (isExternalLink(url)) {
        if (typeof gtag !== 'undefined') {
          gtag('event', 'external_link_click', {
            event_category: 'navigation',
            event_label: url.substring(0, 100), // 截断过长URL
            link_text: this.textContent.substring(0, 50)
          });
        }
      }
    });
  });
})();
</script>
```

## 四、百度统计配置（可选）

### 4.1 获取百度统计代码

1. 注册 [百度统计](https://tongji.baidu.com/)
2. 添加网站并验证所有权
3. 获取统计代码（格式：`hm.src = 'https://hm.baidu.com/hm.js?xxxxxxxx'`）

### 4.2 配置方式

编辑项目根目录 `_config.yml`：

```yaml
baidu_tongji:
  enable: true
  code: 'xxxxxxxx'  # 替换为您的百度统计 ID
```

### 4.3 手动添加代码

在 `themes/Chic/layout/_partial/head.ejs` 中添加：

```html
<!-- 百度统计 -->
<script>
var _hmt = _hmt || [];
(function() {
  var hm = document.createElement("script");
  hm.src = "https://hm.baidu.com/hm.js?xxxxxxxx"; // 替换为您的代码
  var s = document.getElementsByTagName("script")[0];
  s.parentNode.insertBefore(hm, s);
})();
</script>
```

## 六、转化漏斗分析

### 6.1 定义转化漏斗

建议的漏斗模型：

```
首页访问 → 文章列表 → 文章详情 → 阅读完成 → 分享/互动
```

### 6.2 GA4 漏斗配置

在 GA4 中通过「配置」→「漏斗」创建：

1. **步骤 1：首页访问** - 选择「页面浏览」事件，路径包含 `/`
2. **步骤 2：文章列表** - 路径包含 `/archives` 或 `/categories`
3. **步骤 3：文章详情** - 路径匹配 `/:year/:month/:day/:hash/`
4. **步骤 4：阅读完成** - 触发 `article_read_complete` 事件
5. **步骤 5：分享/互动** - 触发 `share_click` 事件

## 九、常见问题

### Q1: GA4 代码添加后不生效？

1. 确认 Measurement ID 正确
2. 清除浏览器缓存
3. 使用 GA4 Debug 模式验证
4. 检查浏览器控制台是否有错误
5. 确认是 `measurement_id`（根配置）还是 `tracking_id`（主题配置）

### Q2: 如何验证追踪是否正常？

1. 安装 [GA4 DebugView](https://chrome.google.com/webstore/detail/google-analytics-debugger/jnkmfdileelhofjcijadephnlnkepbka) 扩展
2. 访问网站并触发事件
3. 在 DebugView 中查看实时事件
4. 等待 24-48 小时在标准报告中查看数据

### Q3: 静态博客能追踪哪些指标？

| 指标 | 支持情况 | 说明 |
|------|----------|------|
| 页面浏览量 (PV) | ✅ 完全支持 | 自动追踪 |
| 独立访客数 (UV) | ✅ 完全支持 | 通过 Cookie 识别 |
| 用户停留时长 | ✅ 通过事件估算 | 基于页面停留时间计算 |
| 跳出率 | ✅ 完全支持 | 自动计算 |
| 滚动完成率 | ⚠️ 需手动配置 | 需添加事件追踪代码 |
| 分享点击 | ⚠️ 需手动配置 | 需添加事件追踪代码 |
| 外部链接点击 | ⚠️ 需手动配置 | 需添加事件追踪代码 |

### Q4: 根配置和主题配置有什么区别？

- **根 _config.yml**：供 Hexo 插件或自定义集成使用
- **主题 _config.yml**：Chic 主题内置的 GA4 支持，直接在主题中处理追踪

建议使用主题配置（更简单），或根据需要选择其一。
