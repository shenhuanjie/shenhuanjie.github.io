## 一、当前阅读体验问题分析

### 1.1 现有样式分析

基于 `media.styl` 和 `post_content.styl` 分析，当前移动端阅读体验存在以下问题：

| 问题类型 | 当前实现 | 影响 |
|---------|---------|-----|
| 字体大小 | 15px (post-content) | 可读性一般 |
| 行距 | 1.9em | 段落区分不明显 |
| 代码块 | 11px, overflow-x: auto | 代码拥挤 |
| 图片 | max-width 100% | 可能超出视口 |
| 表格 | overflow-x: auto | 横向滚动体验差 |

### 1.2 优化目标

| 指标 | 当前值 | 目标值 |
|-----|-------|-------|
| 正文字体大小 | 15px | 16-18px |
| 行高 | 1.9em | 1.8-2.0 |
| 段落间距 | 1em | 1.2-1.5em |
| 代码块字体 | 11px | 13-14px |
| 代码块横向滚动 | 基础 | 流畅滚动 |

## 三、代码块横向滚动优化

### 3.1 当前问题

现有 `post_code.styl` 中的代码块样式：

```stylus
figure.highlight
  margin: 1em 0
  overflow-x: auto    // 基础横向滚动
  position: relative
```

问题：
- 滚动条样式不统一
- 滚动体验不流畅
- 没有惯性滚动
- 触摸滚动区域过小

### 3.2 优化方案

```stylus
// 代码块容器
figure.highlight
  margin: 1.5em -0.8rem     // 扩展到屏幕边缘
  padding: 1em 0.8rem
  background: $light-code-background-color
  border-radius: 8px
  overflow-x: auto
  -webkit-overflow-scrolling: touch  // iOS 惯性滚动
  overflow-y: hidden
  position: relative

  // 自定义滚动条
  &::-webkit-scrollbar
    height: 6px

  &::-webkit-scrollbar-track
    background: rgba(0, 0, 0, 0.1)
    border-radius: 3px

  &::-webkit-scrollbar-thumb
    background: rgba(0, 0, 0, 0.3)
    border-radius: 3px

    &:hover
      background: rgba(0, 0, 0, 0.5)

  // 代码样式
  pre
    margin: 0
    padding: 0
    font-size: 14px        // 增大字体
    line-height: 1.6
    font-family: "SF Mono", Monaco, Consolas, monospace

    code
      padding: 0
      background: transparent
      font-size: inherit
      border-radius: 0

  .gutter
    width: 40px
    padding-right: 10px
    text-align: right
    color: rgba(0, 0, 0, 0.3)
    user-select: none

      color: inherit

    background: $dark-code-background-color

    &::-webkit-scrollbar-thumb
      background: rgba(255, 255, 255, 0.2)

      &:hover
        background: rgba(255, 255, 255, 0.3)
```

### 3.3 代码块横向滚动增强

```javascript
// 代码块滚动增强脚本
document.addEventListener('DOMContentLoaded', () => {
  const codeBlocks = document.querySelectorAll('figure.highlight');

  codeBlocks.forEach(block => {
    // 检测是否需要横向滚动
    const hasOverflow = block.scrollWidth > block.clientWidth;

    if (hasOverflow) {
      block.classList.add('scrollable');

      // 添加触摸滚动提示
      const scrollHint = document.createElement('div');
      scrollHint.className = 'scroll-hint';
      scrollHint.textContent = '← 左右滑动 →';
      block.appendChild(scrollHint);

      // 滚动时隐藏提示
      block.addEventListener('scroll', () => {
        scrollHint.style.opacity = '0';
      });

      // 触摸结束时显示提示
      block.addEventListener('touchend', () => {
        setTimeout(() => {
          scrollHint.style.opacity = '1';
        }, 1500);
      });
    }
  });
});
```

```javascript
// 阅读进度计算
window.addEventListener('scroll', () => {
  const article = document.querySelector('.post-content');
  const progressBar = document.querySelector('.progress-bar');

  if (article && progressBar) {
    const articleTop = article.offsetTop;
    const articleHeight = article.offsetHeight;
    const windowHeight = window.innerHeight;
    const scrollY = window.scrollY;

    const progress = Math.min(100, Math.max(0,
      ((scrollY - articleTop + windowHeight) / articleHeight) * 100
    ));

    progressBar.style.width = `${progress}%`;
  }
});
```

### 6.2 字号调节脚本

```javascript
const FONT_SIZE_KEY = 'article-font-size';
const MIN_SIZE = 14;
const MAX_SIZE = 22;
const DEFAULT_SIZE = 17;

function updateFontSize(size) {
  document.documentElement.style.setProperty('--article-font-size', `${size}px`);
  localStorage.setItem(FONT_SIZE_KEY, size);
}

document.addEventListener('DOMContentLoaded', () => {
  const savedSize = localStorage.getItem(FONT_SIZE_KEY);
  const currentSize = savedSize ? parseInt(savedSize) : DEFAULT_SIZE;
  updateFontSize(currentSize);

  document.getElementById('decrease-font')?.addEventListener('click', () => {
    const current = parseInt(localStorage.getItem(FONT_SIZE_KEY)) || DEFAULT_SIZE;
    if (current > MIN_SIZE) {
      updateFontSize(current - 1);
    }
  });

  document.getElementById('increase-font')?.addEventListener('click', () => {
    const current = parseInt(localStorage.getItem(FONT_SIZE_KEY)) || DEFAULT_SIZE;
    if (current < MAX_SIZE) {
      updateFontSize(current + 1);
    }
  });
});
```

```css

}
```
